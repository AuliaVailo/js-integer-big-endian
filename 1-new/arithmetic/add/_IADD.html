Adds a big endian array to another. Wraps on overflow. |A| >= |B|.

Parameters:

  • r must be a Number.
    (base (radix))

  • a must be an Array.
    (first operand)

  • ai must be a Number.
    (a left)

  • aj must be a Number.
    (a right)

  • b must be an Array.
    (second operand)

  • bi must be a Number.
    (b left)

  • bj must be a Number.
    (b right)

const _IADD = function ( r , a , ai , aj , b , bi , bj ) { let C = 0 ; while ( --bj >= bi ) { const t = a[--aj] + b[bj] + C ; a[aj] = t % r ; C = t >= r ; } while ( --aj >= ai ) { const t = a[aj] + C ; a[aj] = t % r ; C = t >= r ; } } ; exports._IADD = _IADD ;
