Adds 1 to a big endian array.

Parameters:

  • r must be a Number.
    (radix)

  • a must be an Array.
    (first operand)

  • ai must be a Number.
    (a left)

  • aj must be a Number.
    (a right)

const _increment = function ( r , a , ai , aj ) { const _r = r - 1 ; while ( --aj >= ai ) { if ( a[aj] < _r ) { ++a[aj] ; return ; } a[aj] = 0 ; } } ; exports._increment = _increment ;
