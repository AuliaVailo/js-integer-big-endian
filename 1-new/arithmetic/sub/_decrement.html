Subtracts 1 from a big endian array.

Parameters:

  • r must be a Number.
    (radix)

  • a must be an Array.
    (first operand)

  • ai must be a Number.
    (a left)

  • aj must be a Number.
    (a right)

const _decrement = function ( r , a , ai , aj ) { while ( --aj >= ai ) { if ( a[aj] > 0 ) { --a[aj] ; return ; } a[aj] = r - 1 ; } } ; exports._decrement = _decrement ;
