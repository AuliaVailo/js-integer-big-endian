Subtracts B from A, |A| >= |B|. Wraps.

Parameters:

  • r must be an int.
    (base (radix))

  • a must be an array.
    (first operand)

  • ai must be an int.
    (a left)

  • aj must be an int.
    (a right)

  • b must be an array.
    (second operand)

  • bi must be an int.
    (b left)

  • bj must be an int.
    (b right)

const _isub = function ( r , a , ai , aj , b , bi , bj ) { let C = 0 ; while ( --bj >= bi ) { --aj ; const T = C ; C = a[aj] < b[bj] + T ; a[aj] = a[aj] - b[bj] + (C*r - T) ; } while ( C && --aj >= ai ) { const T = C ; C = a[aj] < T ; a[aj] += ( C*r - T ) ; } } ; exports._isub = _isub ;
